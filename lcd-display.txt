#include <Wire.h> 
#include <LiquidCrystal_I2C.h>

//Crear el objeto lcd  direcci√≥n  0x3F y 16 columnas x 2 filas
LiquidCrystal_I2C lcd(0x20,16,2);  //

void setup() {
  // Inicializar el LCD
  lcd.init();
  
  //Encender la luz de fondo.
  lcd.backlight();
  
  // Escribimos el Mensaje en el LCD.
  lcd.print("WELCOME");
   lcd.setCursor(0,1);
 lcd.print("IOT LAB");
 delay(10);
}

void loop() {
  lcd.setCursor(0,1);
  lcd.print("Mubashir");
  delay(10);
}
